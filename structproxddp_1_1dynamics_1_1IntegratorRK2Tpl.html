template<typename _Scalar>
struct proxddp::dynamics::IntegratorRK2Tpl< _Scalar >

Second-order Runge-Kutta integrator.

\begin{eqnarray*} x_{k+1} = x_k \oplus h f(x^{(1)}, u_k),\\ x^{(1)} = x_k \oplus \frac h2 f(x_k, u_k) \end{eqnarray*}
