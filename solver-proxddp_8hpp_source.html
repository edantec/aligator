150  void setup(const Problem &problem) {
151  workspace_ = std::make_unique<Workspace>(problem);
152  results_ = std::make_unique<Results>(problem);
153 
154  Workspace *ws = workspace_.get();
155  prox_penalties_.clear();
156  const std::size_t nsteps = problem.numSteps();
157  for (std::size_t i = 0; i < nsteps; i++) {
158  const StageModel &sm = *problem.stages_[i];
159  prox_penalties_.emplace_back(sm.xspace_, sm.uspace_, ws->prev_xs_[i],
160  ws->prev_us_[i], false);
161  if (i == nsteps - 1) {
162  prox_penalties_.emplace_back(sm.xspace_next_, sm.uspace_,
163  ws->prev_xs_[nsteps],
164  problem.dummy_term_u0, true);
165  }
166  }
167 
168  for (std::size_t i = 0; i < nsteps + 1; i++) {
169  const ProxPenaltyType *penal = &prox_penalties_[i];
170  ws->prox_datas.emplace_back(new ProxData(penal));
171  }
172 
173  assert(prox_penalties_.size() == (nsteps + 1));
174  assert(ws->prox_datas.size() == (nsteps + 1));
175  }
